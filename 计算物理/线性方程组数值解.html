高斯-赛德尔迭代法

原理

高斯-赛德尔法与雅可比迭代的区别在于求解向量的一步，Gauss-Seidel法并不一次求出下一步的解向量，而是每计算出某一代解向量中的一个元素就用它来替换上一代的对应元素，然后求这一代的下一个元素

源代码

由于Gauss-Seidel法与Jacobi法的区别仅在于求解向量的迭代过程，程序流程完全一致，故不再画出程序框图，以下代码亦仅展示 部分

 
xxxxxxxxxx
subroutine calcx()
    xvec=0
    PRINT"(a22$)",'input step number: '
    read(*,*)step
    print *,"-----------------start calculate x----------------------------"
    do m=1,step
        do n=1,ndim
            xvec(n)=DOT_PRODUCT(gmat(n,:),xvec)+gvec(n)
        end do
        print"(i5,9f8.3)",m,xvec(:)
    end do
    print *,"-----------------calculate vecter x done----------------------------"
end subroutine

subroutine calcx()
    xvec=1
    print *,"-----------------start calculate x----------------------------"
    do m=1,step
        do n=1,ndim
            xvec(n)=(1-ome)*xvec(n)+ome*(DOT_PRODUCT(gmat(n,:),xvec)+gvec(n))
        end do
